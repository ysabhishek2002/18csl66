%{
#include<stdio.h>
#include<stdlib.h>
%}

%token num
%left '+' '-'
%left '*' '/'

%%
E: exp {printf("%d\n",$$); exit(0);}
exp: exp '+' exp {$$ = $1+$3;;}
| exp '-' exp {$$ = $1-$3;}
| exp '*' exp {$$ = $1*$3;}
|'('exp')' {$$ = $2;}
| num {$$ = $1;};
| exp '/' exp 
   {
	if($3==0)
	{
		printf("Divide by zero error");
		exit(0);
	}
	else
	{
		$$ = $1/$3;
	}
   }

%%

int yyerror()
{
	printf("Error");
	exit(0);
}

{
	printf("Enter the expression:");
	yyparse();
}
